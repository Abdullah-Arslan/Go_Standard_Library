Aritmetik Operatörler

package main

import "fmt"

func main() {
    x, y := 10, 3
    fmt.Println("x + y =", x+y) // 13
    fmt.Println("x - y =", x-y) // 7
    fmt.Println("x * y =", x*y) // 30
    fmt.Println("x / y =", x/y) // 3  (tam sayı bölme)
    fmt.Println("x % y =", x%y) // 1
}

Atama Operatörleri

package main

import "fmt"

func main() {
    x := 5

    x += 3
    fmt.Println("x += 3 ->", x) // 8

    x -= 2
    fmt.Println("x -= 2 ->", x) // 6

    x *= 4
    fmt.Println("x *= 4 ->", x) // 24

    x /= 6
    fmt.Println("x /= 6 ->", x) // 4

    x %= 3
    fmt.Println("x %= 3 ->", x) // 1

    x &= 3
    fmt.Println("x &= 3 ->", x)

    x |= 2
    fmt.Println("x |= 2 ->", x)

    x ^= 1
    fmt.Println("x ^= 1 ->", x)

    x <<= 1
    fmt.Println("x <<= 1 ->", x)

    x >>= 1
    fmt.Println("x >>= 1 ->", x)
}

Karşılaştırma Operatörleri

package main

import "fmt"

func main() {
    x, y := 10, 20
    fmt.Println("x == y:", x == y) // false
    fmt.Println("x != y:", x != y) // true
    fmt.Println("x > y:",  x > y)  // false
    fmt.Println("x < y:",  x < y)  // true
    fmt.Println("x >= y:", x >= y) // false
    fmt.Println("x <= y:", x <= y) // true
}

Mantıksal Operatörler

package main

import "fmt"

func main() {
    a, b := true, false
    fmt.Println("a && b:", a && b) // false
    fmt.Println("a || b:",  a || b)     // true
    fmt.Println("!a:",      !a)         // false
}

Bit Düzeyinde Operatörler

package main

import "fmt"

func main() {
    x, y := 6, 3 // 6 = 110, 3 = 011 (ikilik)

    fmt.Println("x & y =",  x&y)   // 2 (010)
    fmt.Println("x | y =",  x|y)   // 7 (111)
    fmt.Println("x ^ y =",  x^y)   // 5 (101)
    fmt.Println("x &^ y =", x&^y) // 4 (100)

    fmt.Println("x << 1 =", x<<1) // 12 (1100)
    fmt.Println("x >> 1 =", x>>1) // 3  (011)
}

Pointer (Adres) Operatörleri

package main

import "fmt"

func main() {
    x := 42
    p := &x // adresini al
    fmt.Println("p =", p)   // örn: 0xc00000a0a8
    fmt.Println("*p =", *p) // 42

    *p = 100                // pointer üzerinden değeri değiştir
    fmt.Println("x =", x)   // 100
}

Özel Operatörler

Kanal <-

package main

import "fmt"

func main() {
    ch := make(chan int)
    go func() { ch <- 10 }()
    val := <-ch
    fmt.Println("kanaldan:", val) // 10
}

Slice genişletme ...

package main

import "fmt"

func sum(nums ...int) int {
    s := 0
    for _, n := range nums { s += n }
    return s
}

func main() {
    a := []int{1,2,3,4}
    fmt.Println("sum:", sum(a...)) // 10
}

Artırma / Azaltma (Yalnızca Postfix)

package main

import "fmt"

func main() {
    i := 5
    i++
    fmt.Println("i++ ->", i) // 6
    i--
    fmt.Println("i-- ->", i) // 5
    // Not: ++i veya --i (prefix) Go'da yoktur.
}
